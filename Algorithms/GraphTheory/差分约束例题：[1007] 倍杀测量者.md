# P4926 [1007] 倍杀测量者

## 代码

时间复杂度 $O(nm \log W)$。

```cpp
#include <bits/stdc++.h>
#define memset0(a) memset(a, 0, sizeof a)
using namespace std;
typedef long long ll;
typedef pair<int, double> pii;

const int MAXN = 1000 + 5;
const double EPS = 1e-6;
const double INF = 1e9;

int n, m, t;

struct Edge {
    int o, u, v, w;
    Edge(int o, int u, int v, int w) : o(o), u(u), v(v), w(w) {}
};
vector<Edge> edges;
vector<pii> scores;

vector<pii> G[MAXN];
void addedge(int u, int v, double w) {
    G[u].emplace_back(v, w);
}

double dis[MAXN];
queue<int> q;
bool inq[MAXN];
int cnt[MAXN];
bool SPFA(int s) {
    for (int u = 1; u <= n; u++)
        dis[u] = INF;
    dis[s] = 0;
    q.push(s); inq[s] = 1; cnt[s]++;
    while (!q.empty()) {
        int u = q.front(); q.pop(); inq[u] = 0;
        for (auto [v, w] : G[u]) {
            if (dis[v] > dis[u] + w + EPS) {
                dis[v] = dis[u] + w;
                if (!inq[v]) {
                    q.push(v); inq[v] = 1; cnt[v]++;
                }
                if (cnt[v] == n + 2)
                    return 1;
            }
        }
    }
    return 0;
}
void clear() { // 清空 SPFA 数组
    for (int u = 1; u <= n; u++) G[u].clear();
    memset0(dis);
    while (!q.empty()) q.pop();
    memset0(inq);
    memset0(cnt);
}

bool check(double x) {
    clear();
    for (auto [o, u, v, w] : edges) {
        if (o == 1) {
            if (w - x < EPS)
                continue;
            addedge(u, v, -log(w - x));
        } else if (o == 2)
            addedge(u, v, log(w + x));
    }
    for (auto [u, w] : scores) {
        addedge(0, u, log(w));
        addedge(u, 0, -log(w));
    }
    return SPFA(0);
}

int main() { ios::sync_with_stdio(0); cin.tie(0);
    cin >> n >> m >> t;
    for (int i = 1; i <= m; i++) {
        int o, u, v, w; cin >> o >> u >> v >> w;
        edges.emplace_back(o, u, v, w);
    }
    for (int i = 1; i <= t; i++) {
        int u, w; cin >> u >> w;
        scores.emplace_back(u, w);
    }

    if (!check(0))
        cout << -1 << endl;
    else {
        double l = 0, r = 1e9, ans = -1;
        while (r - l > EPS) {
            double mid = (l + r) / 2;
            if (check(mid)) {
                l = mid + EPS;
                ans = mid;
            } else
                r = mid - EPS;
        }
        cout << fixed << setprecision(10) << ans << endl;
    }
    return 0;
}
```

## 注

使用的是取对数后使用差分约束的方法。
